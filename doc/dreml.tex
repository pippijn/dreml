\newcommand{\newboxenv}[4]
  {\newenvironment{#1}
    {
      \pagebreak[2]
      \ifx\bw\relax
        \definecolor{shadecolor}{gray}{#3}
      \else
        \definecolor{shadecolor}{rgb}{#4}
      \fi
      \setlength{\topsepbackup}{\topsep}
      \setlength{\topsep}{0pt}
      \begin{shaded}
      \begin{#2}
    }{
      \end{#2}
      \end{shaded}
      \setlength{\topsep}{\topsepbackup}
    }
  }
